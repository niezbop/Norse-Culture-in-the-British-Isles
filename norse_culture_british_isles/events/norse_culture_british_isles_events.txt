norse_culture_british_isles.1001 = {
	type = character_event
	title = norse_culture_british_isles.1001.t
	desc = norse_culture_british_isles.1001.desc
	theme = culture_change
	left_portrait = {
		character = scope:northeod_founder
		animation = personality_honorable
	}

	immediate = {
		show_as_tooltip = {
			custom_tooltip = global_culture.custom.merge_culture
			culture:northeod = {
				get_all_innovations_from = scope:old_culture_1
				get_all_innovations_from = culture:anglo_saxon
			}
		}
	}

	option = {
		name = norse_culture_british_isles.1001.a
	}
}
